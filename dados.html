Edge Function — Proxy CORS

Browsers bloqueiam chamadas diretas ao Omie. Cole este código em
Supabase → Edge Functions → New → "omie-proxy": 

// supabase/functions/omie-proxy/index.ts
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
serve(async (req) => {
  const body = await req.json();
  const res = await fetch("https://app.omie.com.br/api/v1/" + body.endpoint, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body.payload),
  });
  const data = await res.json();
  return new Response(JSON.stringify(data), {
    headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" },
  });
});
